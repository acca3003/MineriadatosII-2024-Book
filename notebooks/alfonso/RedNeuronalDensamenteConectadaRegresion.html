Normalizacin

In[24]:
# Realizamos la "Normalizacin" restando la media y dividiendo por la desviacin tpica
# Ahora tendremos valores (-x,x) alredor de 0, pero en general pequeos
mean = train_data.mean(axis=0)
train_data -= mean
std = train_data.std(axis=0)
train_data /= std
test_data -= mean
test_data /= std

Creacin del modelo

In[25]:
# Creamos el modelo

# Inicializamos el API Secuencial de capas
model = keras.Sequential([
        # Aadimos capa de entrada con las 13 variables explicativas
        keras.Input(shape=(13,)),
        # Aadimos capa densamente conectada con 64 neuronas y activacin relu
        layers.Dense(64, activation="relu"),
        # Aadimos capa densamente conectada con 64 neuronas y activacin relu
        layers.Dense(64, activation="relu"),
        # Aadimos capa de salida densamente conectada con 1 neurona y activacin lineal (para regresin)
        layers.Dense(1)
    ])

# Mostramos el Modelo creado
model.summary()

# Compilamos el modelo definiendo el optimizador, funcin de prdida y mtrica
# RMSProp, mse, mae
model.compile(optimizer="rmsprop", loss="mse", metrics=["mae"])
